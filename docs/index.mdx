# Action Sack
> Collection of React user-experience containers for common interactions

### TempValue

import {TempValue} from '../src/TempValue'

<TempValue initialValue="Mike">
  {({ hasChanged, value, onInputChange, onSubmit, onCancel }) => (
    <form onSubmit={e => e.preventDefault()}>
      <input value={value || ''} onChange={onInputChange} />
      <button disabled={!hasChanged} onClick={onSubmit}>Submit</button>
      <button disabled={!hasChanged} onClick={onCancel}>Cancel</button>
    </form>
  )}
</TempValue>

### CopyButton

import {CopyButton} from '../src/CopyButton'

<CopyButton text="Copied!">
  {({ copied }) => <button>{copied ? 'Copied!': 'Copy'}</button>}
</CopyButton>

### AsyncButton

import {AsyncButton} from '../src/AsyncButton'

<AsyncButton onClick={() => new Promise(res => setTimeout(function () { res() }, 1000))}>
  {({ loading, onClick }) => <button onClick={onClick}>{loading ? 'Saving...': 'Save'}</button>}
</AsyncButton>
